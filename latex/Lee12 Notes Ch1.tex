\hpart{Generalities of Smooth Manifolds}
\section{Smooth Manifold}
\begin{definition}
    A topological space $M$ is a \hdef{topological $n$-manifold} if the following properties hold.
    \begin{enumerate}
        \item For every $x\in M$, there exists a neighborhood ${U}_{x}$ of $x$ homeomorphic to an open subset of ${\R}^{n}$.
        \item The space $M$ is second-countable.
        \item The space $M$ is Hausdorff.
    \end{enumerate}
\end{definition}
\begin{remark}
    The empty set $\es$ is vacuously a topological $n$-manifold. We shall ignore this case.
\end{remark}
\begin{proposition}
    Let $M$ be a topological space. Then the following are equivalent.
    \begin{enumerate}
        \item For every $x\in M$, there exists a neighborhood ${U}_{x}$ of $x$ homeomorphic to an open subset of ${\R}^{n}$.
        \item For every $x\in M$, there exists a neighborhood ${U}_{x}$ of $x$ homeomorphic to an open ball of ${\R}^{n}$.
        \item For every $x\in M$, there exists a neighborhood ${U}_{x}$ of $x$ such that ${U}_{x}\iso{\R}^{n}$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    (2$\Lra$3) Any open ball in ${\R}^{n}$ is homeomorphic to ${\R}^{n}$. (1$\Ra$3) Fix $x\in M$. Let $\vp:{U}_{x}\to V$ be the homeomorphism. Then there exists an open ball centered at $\vp(x)$ of sufficiently small radius such that $B(\vp(x))\sub V$. Let $U'={\vp}^{-1}(B(\vp(x)))$. Since $\vp$ is continuous, $U'$ is the desired neighborhood of $x$. (3$\Ra$1) Trivial.
\end{proof}
\begin{proposition}
    If $m\ne n$, then any topological $m$-manifold is not homeomorphic to any topological $n$-manifold.
\end{proposition}
\par
A self-contained proof will be shown later.
\begin{definition}
    Let $M$ be a topological $n$-manifold. A \hdef{chart} on $M$ is a pair $(U,\vp)$ where $U\sub M$ is open and $\vp:U\to V$ is a homeomorphism onto an open subset $V\sub{\R}^{n}$.
\end{definition}
\par
For every $p\in M$, there exists a chart $(U,\vp)$ such that $p\in U$. If $\vp(p)=0$, then we say $(U,\vp)$ is \hdef{centered at} $p$.
\begin{definition}
    Let $(U,\vp)$ be a chart on a topological manifold $M$. The set $U$ is called a \hdef{coordinate domain}. If $\vp(U)$ is an open ball, then $U$ is called a \hdef{coordinate ball}. If $\vp(U)$ is an open cube, then $U$ is called a \hdef{coordinate cube}. The map $\vp$ is called the \hdef{coordinate map}. For every $p\in U$, rewrite $\vp(p)=({x}_{1}(p),\dots,{x}_{n}(p))$. The coordinates ${x}_{i}(p)$ are called \hdef{local coordinates} on $U$.
\end{definition}
\begin{example}
    Let $U\sub{\R}^{n}$ be open and let $f:U\to{\R}^{k}$ be continuous. The \hdef{graph} of $f$ is $\Ga(f)=\{(x,y)\mid x\in U,y=f(x)\}$ endowed with the subspace topology, so it is Hausdorff and second countable. Let $\pi:{\R}^{n}\times{\R}^{k}\to{\R}^{n}$ be the natural projection, then $\pi{|}_{\Ga(f)}$ is a homeomorphism onto its image. Hence $\Ga(f)$ is a topological $n$-manifold.
\end{example}
\begin{example}
    Recall that ${S}^{n}=\{x\in{\R}^{n+1}\mid|x|=1\}$. Equip ${S}^{n}$ with the subspace topology, then ${S}^{n}$ is Hausdorff and second-countable. Let ${U}_{i}^{+}=\{({x}_{1},\dots,{x}_{n+1})\mid{x}_{i}>0\}\sub{\R}^{n+1}$ and similarly define ${U}_{i}^{-}$, so $\{{U}_{i}^{\pm}\}$ covers ${S}^{n}$. Define $f:{B}^{n}\to\R$ by $f:u\mt\sqrt{1-{|u|}^{2}}$, so $f$ is continuous. If $({x}_{1},\dots,{x}_{n+1})\in{S}^{n}$, then $({x}_{1},\dots,\wh{{x}_{i}},\dots,{x}_{n+1})\in{B}^{n}$. We have $\Ga(f)={U}_{i}^{+}\cap{S}^{n}$ and $\Ga(-f)={U}_{i}^{-}\cap{S}^{n}$ for some ${U}_{i}$. Since $\Ga(f)$ and $\Ga(-f)$ are topological $n$-manifolds, ${S}^{n}$ is a topological $n$-manifold.
\end{example}
\begin{example}
    The $n$-dimensional \hdef{real projective space} $\RP{n}$ is the set of all 1-dimensional linear subspaces of ${\R}^{n+1}$. Equivalently, we may define $\RP{n}$ via the quotient map $\pi:{\R}^{n+1}\sm\{0\}\to\RP{n}$ defined by $\pi:x\mt\cyc{x}$. Define $\wt{{U}_{i}}=\{({x}_{1},\dots,{x}_{n+1})\mid{x}_{i}\ne 0\}\sub{\R}^{n+1}\sm\{0\}$ and define ${U}_{i}=\pi(\wt{{U}_{i}})$. Fix $i$. Trivially $\wt{{U}_{i}}$ is open and saturated with respect to $\pi$. The restriction of a quotient map to an open saturated set is continuous, so $\pi{|}_{\wt{{U}_{i}}}$ is a continuous quotient map. Now define ${\vp}_{i}:{U}_{i}\to{\R}^{n}$ by ${\vp}_{i}:\cyc{({x}_{1},\dots,{x}_{n+1})}\mt({x}_{1}/{x}_{i},\dots,\wh{{x}_{i}/{x}_{i}},\dots,{x}_{n+1}/{x}_{i})$. This map is well-defined. The map ${\vp}_{i}\comp\pi$ on each coordinate is continuous, hence ${\vp}_{i}\comp\pi$ is continuous on $\wt{{U}_{i}}$. By the universal property for quotient spaces, ${\vp}_{i}$ is continuous.
    \begin{center}
        \begin{tikzcd}
            {{\R}^{n}} & {{U}_{i}} \\
            & {\wt{{U}_{i}}}
            \arrow["{{\vp}_{i}}"', from=1-2, to=1-1]
            \arrow["{{\vp}_{i}\comp\pi}", from=2-2, to=1-1]
            \arrow["{\pi}"', from=2-2, to=1-2]
        \end{tikzcd}
    \end{center}
    Consider ${\psi}_{i}:{\R}^{n}\to{U}_{i}$ defined by ${\psi}_{i}:({y}_{1},\dots,{y}_{n})\mt\cyc{{y}_{1},\dots,{y}_{i-1},1,{y}_{i},\dots,{y}_{n}}$, and ${\psi}_{i}$ is the continuous inverse of ${\vp}_{i}$. Since $\{{U}_{i}\}$ is an open cover of $\RP{n}$, $\RP{n}$ is locally Euclidean of dimension $n$. Pick $\cyc{x},\cyc{y}\in\RP{n}$. Suppose $\cyc{x}\in{U}_{i}$ and $\cyc{y}\in{U}_{j}$. Since ${U}_{i}\iso{\R}^{n}$, ${U}_{i}$ is Hausdorff, so the case $i=j$ is trivial. Let $i\ne j$. Let $r=\min\{d(x/|x|,y/|y|),d(x/|x|,-y/|y|)\}/3$. Then $\cyc{x}\in\pi({B}_{r}(x/|x|))$, $\cyc{y}\in\pi({B}_{r}(y/|y|))$, and $\pi({B}_{r}(x/|x|))\cap\pi({B}_{r}(y/|y|))=\es$. Hence $\RP{n}$ is Hausdorff. Since each ${U}_{i}$ is second-countable and $\{{U}_{i}\}$ is a finite open cover of $\RP{n}$, $\RP{n}$ is second-countable. Hence $\RP{n}$ is a topological $n$-manifold.
\end{example}
\begin{proposition}
    The space $\RP{n}$ is compact.
\end{proposition}
\begin{proof}
   For any $\cyc{x}\in\RP{n}$, $|x/|x||=1\in{S}^{n}$, so $\pi{|}_{{S}^{n}}$ is surjective. Since ${S}^{n}$ is compact, $\RP{n}$ is compact.
\end{proof}
\begin{example}
    Let ${M}_{1},\dots,{M}_{k}$ be topological manifolds with $\dim{M}_{i}={n}_{i}$. The product space $M={M}_{1}\times\cdots\times{M}_{k}$ is Hausdorff and second-countable. For every $({p}_{1},\dots,{p}_{k})\in M$, there exist charts $({U}_{i},{\vp}_{i})$ in each ${M}_{i}$. Define $\vp:{U}_{1}\times\cdots\times{U}_{k}\to{\vp}_{1}({U}_{1})\times\cdots\times{\vp}_{k}({U}_{k})$ by $\vp:({x}_{1},\dots,{x}_{k})\mt({\vp}_{1}({x}_{1}),\dots,{\vp}_{k}({x}_{k}))$ and the image is open in ${\R}^{{\sum}_{1}^{k}{n}_{i}}$. Since each ${\vp}_{i}$ is a homeomorphism, $\vp$ is a homeomorphism. Hence $M$ is a topological $({\sum}_{1}^{k}{n}_{i})$-manifold. 
\end{example}
\begin{example}
    The \hdef{n-torus} is defined to be the product space ${T}^{n}={S}^{1}\times\cdots\times{S}^{1}$ of $n$ copies of ${S}^{1}$. Hence ${T}^{n}$ is a topological $n$-manifold.
\end{example}
\begin{proposition}
    Every topological manifold $M$ has a countable basis of precompact coordinate balls.
\end{proposition}
\begin{proof}
    Let $\dim M=n$. Since $M$ is second countable, $M$ admits a countable cover $\{{U}_{i}\}$. Fix a chart $({U}_{i},{\vp}_{i})$. Consider the collection of balls ${B}_{r}(x)$, where $r\in\Q$ and $x\in{\Q}^{n}$, such that $\wb{{B}_{r}(x)}\sub{\vp}_{i}({U}_{i})$, then $\{{B}_{r}(x)\}$ is a precompact basis for ${\vp}_{i}({U}_{i})$. The collection $\{{\vp}_{i}^{-1}({B}_{r}(x))\}$ is a countable basis of ${U}_{i}$. Since $\wb{{B}_{r}(x)}$ is compact in ${\R}^{n}$, ${\vp}_{i}^{-1}(\wb{{B}_{r}(x)})$ is compact in ${U}_{i}$, Since $M$ is Hausdorff, the closure of ${\vp}_{i}^{-1}({B}_{r}(x))$ in ${U}_{i}$ is its closure in $M$. Hence the union of $\{{\vp}_{i}^{-1}({B}_{r}(x))\}$ is a precompact basis for $M$.
\end{proof}
\begin{proposition}
    Let $M$ be a topological manifold. Then the following properties hold.
    \begin{enumerate}
        \item The manifold $M$ is locally path-connected.
        \item The manifold $M$ is connected if and only if it is path-connected.
        \item The components of $M$ are the same as its path components.
        \item The manifold $M$ has countably many components and each of which is an open subset of $M$ and a connected topological manifold.
    \end{enumerate}
\end{proposition}
\begin{proof}
    (\rom1) Open balls are path-connected. Precompact coordinate balls are path-connected, hence $M$ is locally path-connected. (\rom2) Path-connected spaces are connected. By (\rom1), $M$ is locally path-connected. Since $M$ is connected, $M$ is path-connected. (\rom3) Since $M$ is locally path-connected, the path-components are open. Ppath-components partition $M$, so they are clopen. Every path-component is a subset of a connected component. Since every component is connected, it cannot be partitioned into multiple clopen subsets, so each component contains exactly one path-component. (\rom4) Every component of $M$ is open. All components cover $M$, so there exists a countable subcover. Since components are disjoint, there are countably many components. Each component is a subspace of $M$, so it is second-countable and Hausdorff, hence a connected topological manifold.
\end{proof}
\begin{proposition}
    Every topological manifold is locally compact.
\end{proposition}
\begin{proof}
    A topological manifold is Hausdorff and admits a basis of precompact sets, hence it is locally compact.
\end{proof}
\begin{definition}
    Let $M$ be a topological space. Let $X\sub\ps(M)$. We say $X$ is \hdef{locally finite} if each $p\in M$ has a neighborhood that intersects at most finitely many sets in $X$.
\end{definition}
\begin{proposition}
    Let $M$ be a topological space and let $X$ be locally finite. Then the following properties hold.
    \begin{enumerate}
        \item The set $\{\wb{{X}_{i}}\mid{X}_{i}\in X\}$ is locally finite.
        \item The set $\wb{\Un{X}_{i}}=\Un\wb{{X}_{i}}$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    (\rom1) Fix $p\in M$. Since $X$ is locally finite, there exists neighborhood $U$ of $p$ such that it intersects ${X}_{1},\dots,{X}_{n}\in X$. Suppose there exists $y\in U\cap\wb{{X}_{m}}$, then any neighborhood of $y$ intersects ${X}_{m}$, so ${X}_{m}$ is one of ${X}_{1},\dots,{X}_{n}$, hence $U$ intersects at most $n$ closures. (\rom2) Trivially $\Un\wb{{X}_{i}}\sub\wb{\Un{X}_{i}}$. Let $p\in\wb{\Un{X}_{i}}$ and suppose $p\notin\Un\wb{{X}_{i}}$. Since $X$ is locally finite, there exists ${U}_{p}$ that only intersects ${X}_{1},\dots,{X}_{n}$. Since $p\notin\Un\wb{{X}_{i}}$, $W={U}_{p}\cap{\In}_{1}^{n}(M\sm\wb{{X}_{k}})$ is a neighborhood of $p$. Since $W\sub{U}_{p}$ and $W\sub M\sm\wb{{X}_{k}}$, $W\cap{X}_{k}=\es$, so $W\cap\Un{X}_{i}=\es$, a contradiction.
\end{proof}
\begin{definition}
    Let $M$ be a topological space and let $U$ be a cover of $M$. A cover $V$ of $M$ is a \hdef{refinement} of $U$ if for each ${V}_{i}\in V$, there exists a ${U}_{j}\in U$ such that ${V}_{i}\sub{U}_{j}$. We say $M$ is \hdef{paracompact} if every open cover of $M$ admits an open locally finite refinement.
\end{definition}
\begin{proposition}
    Let $M$ be a topological manifold. For every basis $B$ of $M$ and open cover $X$, there exists a countable locally finite refinement of $X$ that consists of basis elements in $B$.
\end{proposition}
\begin{proof}
    Let $\{{K}_{j}\}$ be an exaustion of $M$ by compact sets. Define ${V}_{j}={K}_{j+1}\sm\Int({K}_{j})$ and ${W}_{j}=\Int({K}_{j+2})\sm{K}_{j-1}$, then ${V}_{j}\sub{W}_{j}$ is compact. For every $x\in{V}_{j}$, there exists ${U}_{x}\in X$ with $x\in{U}_{x}$. Since $B$ is a basis, there exists ${B}_{x}\in B$ with $x\in{B}_{x}$ such that ${B}_{x}\sub{U}_{x}\cap{W}_{j}$. The collection $\{{B}_{x}\}$ is an open cover of ${V}_{j}$, since ${V}_{j}$ is compact, there exists a finite subcover $\{{B}_{{x}_{j}}\}$. The union $\Un\{{B}_{{x}_{j}}\}$ is therefore a countable refinement of $X$. Notice that ${W}_{i}$ intersects ${W}_{j}$ if and only if $|i-j|\le 2$, so $\Un\{{B}_{{x}_{j}}\}$ is locally finite.
\end{proof}
\begin{proposition}
    The fundamental group of a topological manifold is countable.
\end{proposition}
\begin{proof}
    Let $M$ be a topological manifold. Pick a countable basis $B$ of coordinate balls of $M$. For every ${B}_{1},{B}_{2}\in B$, ${B}_{1}\cap{B}_{2}$ has at most countably many path-components. For every path-component, fix a point ${x}_{i}$, and let $X$ be the collection of those ${x}_{i}$. Since $B$ is countable, $X$ is countable. For every $x,y\in X$ and $U\in B$, fix a path ${h}_{x,y}^{U}$ joining $x$ and $y$ that entirely lies in $U$. Let $\Om$ be the set of all finite concatenation of such paths, so $\Om$ is also countable. Given any loop $f:[0,1]\to M$ based at $p\in M$, the set $\{{f}^{-1}(U)\mid U\in B\}$ is an open cover of $[0,1]$. Since $[0,1]$ is compact, by the Lebesgue's number lemma, there exists a partition $0={a}_{0}<{a}_{1}<\cdots<{a}_{k}=1$ such that each $[{a}_{i-1},{a}_{i}]$ is mapped by $f$ entirely into some ${U}_{i}\in B$. For each $i\in\{1,\dots,k-1\}$, $f({a}_{i})$ lies in ${U}_{i}\cap{U}_{i+1}$, so $f({a}_{i})$ belongs to a specific path-component of ${U}_{i}\cap{U}_{i+1}$. Let ${x}_{i}\in X$ be the chosen reference point for that path-component, and let ${g}_{i}$ be a path from $f({a}_{i})$ to ${x}_{i}$ lying entirely within that path-component. Define ${x}_{0}={x}_{k}=p$ and define ${g}_{0}={g}_{k}={c}_{p}$, the constant path. Let ${f}_{i}=f{|}_{[{a}_{i-1},{a}_{i}]}$, then $f\simeq({f}_{1}*{g}_{1})*({g}_{1}^{-1}*{f}_{2}*{g}_{2})*\cdots*(g_{k-1}^{-1}*{f}_{k})$. Notice that ${g}_{i-1}^{-1}*{f}_{i}*{g}_{i}$ lies in ${U}_{i}$, since ${U}_{i}$ is simply connected, any paths with the same endpoints are homotopic. Hence $f$ is homotopic to a path in $\Om$ and $[f]\in\Om$. Since $\Om$ is countable, ${\pi}_{1}(M,p)$ is countable.
\end{proof}
\begin{definition}
    Let $U\sub{\R}^{n}$ and $V\sub{\R}^{m}$ be open. A function $F:U\to V$ is a \hdef{smooth function} if each component function of $F$ has continuous partial derivatives of all orders. A bijective smooth map is a \hdef{diffeomorphism} if its inverse map is smooth.
\end{definition}
\par
We sometimes use \hdef{${C}^{\infty}$ functions} to denote smooth functions.
\begin{definition}
    Let $M$ be a topological $n$-manifold. Let $(U,\vp)$ and $(V,\psi)$ be charts. The \hdef{transition map} from $\vp$ to $\psi$ is the map $\psi\comp{\vp}^{-1}:\vp(U\cap V)\to\psi(U\cap V)$ defined by $\psi\comp{\vp}^{-1}:\vp(x)\mt\psi(x)$. Two charts are \hdef{smoothly compatible} if the transition map is a diffeomorphism.
\end{definition}
\begin{center}
    \begin{tikzcd}
        & {U\cap V} & \\
        {\vp(U\cap V)} && {\psi(U\cap V)}
        \arrow["\vp"', from=1-2, to=2-1]
        \arrow["\psi", from=1-2, to=2-3]
        \arrow["{\psi\comp{\vp}^{-1}}"', shift right, from=2-1, to=2-3]
        \arrow["{\vp\comp{\psi}^{-1}}"', shift right, from=2-3, to=2-1]
    \end{tikzcd}
\end{center}
\par
If $U\cap V=\es$, the transition map $\psi\comp{\vp}^{-1}$ is vacuously a diffeomorphism.
\begin{definition}
    Let $M$ be a topological manifold. An \hdef{atlas} $\ca{A}$ for $M$ is a collection $\{({U}_{i},{\vp}_{i})\}$ of charts such that $\Un{U}_{i}=M$. An atlas is a \hdef{smooth atlas} if any two charts in the collection are smoothly compatible. A smooth atlas $\ca{A}$ for $M$ is \hdef{maximal} if it is not properly contained in any smooth atlas. 
\end{definition}
\begin{definition}
    Let $M$ be a topological manifold. A maximal smooth atlas $\ca{A}$ for $M$ is called a \hdef{smooth structure} on $M$. A pair $(M,\ca{A})$ is called a \hdef{smooth manifold}.
\end{definition}
\begin{proposition}
    Let $M$ be a topological manifold. Then the following properties hold.
    \begin{enumerate}
        \item Every smooth atlas $\ca{A}$ for $M$ is contained in a unique maximal smooth atlas, called the smooth structure determined by $\ca{A}$.
        \item Two smooth atlases for $M$ determine the same smooth structure if and only if their union is a smooth atlas.
    \end{enumerate}
\end{proposition}
\begin{proof}
    (\rom1) Let $\ca{A}$ be a smooth atlas for $M$. Consider the collection $\wb{\ca{A}}$ of all charts that are compatible with every chart in $\ca{A}$. Clearly, $\ca{A}$ is contained in $\wb{\ca{A}}$, so $\wb{\ca{A}}$ is an atlas for $M$. Let $(U,\vp)$ and $(W,\theta)$ be two charts in $\wb{\ca{A}}$. For every $x\in U\cap W$, there exists a chart $(V,\psi)\in\ca{A}$ that contains $x$. On the intersection $U\cap V\cap W$, the transition map $\vp\comp{\psi}^{-1}\comp(\psi\comp{\theta}^{-1})=\vp\comp{\theta}^{-1}$ is a composition of diffeomorphisms, hence a diffeomorphism. Since the map is smooth on $U\cap V\cap W$, it is smooth at $x$. Because $x$ was arbitrary, the map is smooth everywhere on $U\cap W$. Hence, $\wb{\ca{A}}$ is a smooth atlas. If $\wb{\ca{A}}\sub\ca{B}$, where $\ca{B}$ is a smooth atlas, then any chart in $\ca{B}$ is compatible with all charts in $\ca{A}$. Hence $\ca{B}\sub\wb{\ca{A}}$ and $\wb{\ca{A}}$ is maximal. The uniqueness of $\wb{\ca{A}}$ is trivial. (\rom2) Let $\ca{B}$ be a smooth atlas for $M$. ($\Ra$) If $\wb{\ca{B}}=\wb{\ca{A}}$, then charts in $\ca{B}$ are compatible with charts in $\ca{A}$, so $\ca{A}\cup\ca{B}$ is smooth. ($\La$) If $\ca{A}\cup\ca{B}$ is smooth, then every chart in $\ca{A}$ is compatible with all charts in $\ca{B}$, so $\ca{A}\sub\wb{\ca{B}}$ and $\wb{\ca{A}}\sub\wb{\ca{B}}$. Similarly, $\wb{\ca{B}}\sub\wb{\ca{A}}$. Hence $\wb{\ca{A}}=\wb{\ca{B}}$.
\end{proof}
\par
If all transition maps are $k$-differentiable (${C}^{k}$) diffeomorphisms instead of ${C}^{\infty}$ diffeomorphisms, we obtain a ${C}^{k}$ structure. A manifold with a ${C}^{k}$ structure is called a \hdef{${C}^{k}$ manifold}.
\begin{definition}
    Let $(M,\ca{A})$ be a smooth manifold. Every chart $(U,\vp)\in\ca{A}$ is called a \hdef{smooth chart}. The set $U$ is called a \hdef{smooth coordinate domain} and the map $\vp$ is called a \hdef{smooth coordinate map}. If $\vp(U)$ is a ball, then $U$ is called a \hdef{smooth coordinate ball}.
\end{definition}
\begin{definition}
    Let $M$ be a smooth $n$-manifold. A set $B\sub M$ is a \hdef{regular coordinate ball} if there exists a smooth chart $(B',\vp)$ with $\wb{B}\sub B'$ such that for some positive reals $r<r'$, $\vp(B)={B}_{r}(0)$, $\vp(\wb{B})=\wb{{B}_{r}(0)}$, and $\vp(B')={B}_{r'}(0)$.
\end{definition}
\begin{proposition}
    Every smooth manifold has a countable basis of regular coordinate balls.
\end{proposition}
\begin{proof}
    Let $M$ be a smooth manifold. Since $M$ is second-countable, it admits a countable atlas $\{({U}_{i},{\vp}_{i})\}$. For each $i$, consider the collection of balls ${B}_{r}(a)\sub{\vp}_{i}({U}_{i})$ with $a\in{\Q}^{n}$ and $r\in{\Q}_{>0}$ such that there exists $r'\in{\Q}_{>0}$ with $r<r'$ and $\wb{{B}_{r}(a)}\sub{B}_{r'}(a)\sub{\vp}_{i}({U}_{i})$. This collection is countable. For each such ball, define ${B}_{i,a,r}={\vp}_{i}^{-1}({B}_{r}(a))$ and ${B}_{i,a,r}'={\vp}_{i}^{-1}({B}_{r'}(a))$. Then $({B}_{i,a,r}',{\vp}_{i}{|}_{{B}_{i,a,r}'})$ is a smooth chart. We have ${\vp}_{i}({B}_{i,a,r})={B}_{r}(a)$. Since $\wb{{B}_{r}(a)}$ is compact in ${\R}^{n}$, ${\vp}_{i}^{-1}(\wb{{B}_{r}(a)})$ is compact in $M$. Since $M$ is Hausdorff, ${\vp}_{i}^{-1}(\wb{{B}_{r}(a)})$ is closed in $M$, so ${\vp}_{i}(\wb{{B}_{i,a,r}})=\wb{{B}_{r}(a)}$, and ${\vp}_{i}({B}_{i,a,r}')={B}_{r'}(a)$. Hence ${B}_{i,a,r}$ is a regular coordinate ball. Let $p\in M$ and let $W\sub M$ be a neighborhood of $p$. Choose a chart $({U}_{i},{\vp}_{i})$ with $p\in{U}_{i}$. Then ${\vp}_{i}(W\cap{U}_{i})$ is open in ${\R}^{n}$. There exists a ball ${B}_{r}(a)$ such that ${\vp}_{i}(p)\in{B}_{r}(a)$ and $\wb{{B}_{r}(a)}\sub{\vp}_{i}(W\cap{U}_{i})$. Pick $r'>r$ such that $\wb{{B}_{r}(a)}\sub{B}_{r'}(a)\sub{\vp}_{i}(W\cap{U}_{i})\cap {\vp}_{i}({U}_{i})$. Then $p\in{B}_{i,a,r}\sub W$. Hence the collection of all ${B}_{i,a,r}$ forms a countable basis of $M$.
\end{proof}
\begin{example}
    Let $M$ be a topological 0-manifold. For every $p\in M$, there exists a chart $(U,\vp)$ and $U\iso\{0\}$, so $U=\{p\}$ and $M$ is discrete. Since $M$ is second-countable, it has a countable basis, so there are countably many singletons in $M$, hence $M$ is countable. The set of all charts $\ca{A}$ is an atlas for $M$. Pick charts $(\{p\},\vp)$ and $(\{q\},\psi)$. If $p\ne q$, then the charts are vacuously compatible. If $p=q$, then $\psi\comp{\vp}^{-1}:\{0\}\mt\{0\}$ is the identity map, hence a diffeomorphism. Therefore $(M,\ca{A})$ is a smooth manifold.
\end{example}
\begin{example}
    A topological manifold can admit different smooth structures. The space ${\R}^{n}$ is a topological $n$-manifold covered by the atlas $\{({\R}^{n},{\id}_{{\R}^{n}})\}$. The smooth structure determined by this atlas is called the \hdef{standard smooth structure} on ${\R}^{n}$. Now consider the homeomorphism $\vp:\R\to\R$ with $\vp:x\mt{x}^{3}$. Cover $\R$ by the atlas $\{(\R,\vp)\}$. The transition map ${\id}_{\R}\comp{\vp}^{-1}(y)={y}^{1/3}$ is not smooth at 0. Hence the smooth structure determined by $\{(\R,\vp)\}$ is not the same as the standard smooth structure on $\R$.
\end{example}
\begin{example}
    Let $V$ be a finite-dimensional vector space over $\R$. Any norm on $V$ induces a metric and all norms induce the same metric topology. Let $\{{e}_{1},\dots,{e}_{n}\}$ be a basis of $V$. The map ${f}_{e}:{\R}^{n}\to V$ defined by ${f}_{e}:({x}_{1},\dots,{x}_{n})\mt\sum{x}_{i}{e}_{i}$ is a homeomorphism, so $(V,{f}_{e}^{-1})$ is a chart. \begin{theorem}[chart lemma]
    Let $M$ be a set. Let $\{{U}_{\al}\}\sub\ps(M)$ with maps ${\vp}_{\al}:{U}_{\al}\to{\R}^{n}$. Then $M$ is a smooth $n$-manifold with the smooth structure $\{({U}_{\al},{\vp}_{\al})\}$ if the following properties hold.
    \begin{enumerate}
        \item For every $\al$, ${\vp}_{\al}$ is a bijection onto an open subset of ${\R}^{n}$.
        \item For every $\al$ and $\be$, ${\vp}_{\al}({U}_{\al}\cap{U}_{\be})$ and ${\vp}_{\be}({U}_{\al}\cap{U}_{\be})$ are open.
        \item If ${U}_{\al}\cap{U}_{\be}\ne\es$, then the map ${\vp}_{\be}\comp{\vp}_{\al}^{-1}:{\vp}_{\al}({U}_{\al}\cap{U}_{\be})\to{\vp}_{\be}({U}_{\al}\cap{U}_{\be})$ is smooth.
        \item There exist countably many ${U}_{\al}$ that cover $M$.
        \item Let $p,q\in M$. If $p\ne q$, then either there exists ${U}_{\al}$ such that $p,q\in{U}_{\al}$ or there exist ${U}_{\al}$ and ${U}_{\be}$ such that $p\in{U}_{\al}$, $q\in{U}_{\be}$, and ${U}_{\al}\cap{U}_{\be}=\es$.
    \end{enumerate}
\end{theorem}
\begin{proof}
    Define $\{{\vp}_{\al}^{-1}(V)\}$, where $V\sub{\vp}_{\al}({U}_{\al})$ is open, as a basis of the topology. Let $p\in{\vp}_{\al}^{-1}(V)\cap{\vp}_{\be}^{-1}(W)$. By (\rom1) and (\rom3), ${({\vp}_{\be}\comp{\vp}_{\al}^{-1})}^{-1}(W)\sub{\vp}_{\al}({U}_{\al}\cap{U}_{\be})$, and by (\rom2), ${({\vp}_{\be}\comp{\vp}_{\al}^{-1})}^{-1}(W)$ is open, so $V'=V\cap{({\vp}_{\be}\comp{\vp}_{\al}^{-1})}^{-1}(W)$ is open in ${\R}^{n}$. We have $p\in{\vp}_{\al}^{-1}(V')$. This basis induces a topology on $M$. Clearly (\rom4) implies $M$ is second-countable and (\rom5) implies $M$ is Hausdorff. Hence by (\rom3), $M$ is a smooth $n$-manifold.
\end{proof}

\begin{definition}
    Let $M$ be a topological $n$-manifold with boundary. A \hdef{chart} is a pair $(U,\vp)$, where $U\sub M$ is open and $\vp(U)$ is homeomorphic to either an open subset of ${\R}^{n}$ or an open subset of ${\bb{H}}^{n}$. We say $(U,\vp)$ is an \hdef{interior chart} if $\vp(U)$ is open in ${\R}^{n}$. We say $(U,\vp)$ is a \hdef{boundary chart} if $\vp(U)$ is open in ${\bb{H}}^{n}$ and $\vp(U)\cap\pa{\bb{H}}^{n}\ne\es$. A boundary chart with image of the form ${B}_{r}(x)\cap{\bb{H}}^{n}$, where $x\in\pa{\bb{H}}^{n}$, is called a \hdef{coordinate half-ball}.
\end{definition}
\begin{definition}
    Let $M$ be a topological $n$-manifold with boundary. A point $p\in M$ is an \hdef{interior point} if it is in the domain of an interior chart. The point is a \hdef{boundary point} if it is in the domain of a boundary chart $(U,\vp)$ such that $\vp(p)\in\pa{\bb{H}}^{n}$. The \hdef{boundary} of $M$, denoted $\pa M$, is the set of all boundary points in $M$. The \hdef{interior} of $M$, denoted $\Int(M)$, is the set of all interior points in $M$.
\end{definition}
